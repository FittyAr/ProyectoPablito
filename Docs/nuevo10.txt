# 🧱 1. DbContext

```csharp
public class AppDbContext : DbContext
{
    public DbSet<Movimiento> Movimientos => Set<Movimiento>();
    public DbSet<Cliente> Clientes => Set<Cliente>();
    public DbSet<Trabajo> Trabajos => Set<Trabajo>();
    public DbSet<Factura> Facturas => Set<Factura>();
    public DbSet<PagoFactura> PagosFactura => Set<PagoFactura>();
    public DbSet<Empleado> Empleados => Set<Empleado>();
    public DbSet<Liquidacion> Liquidaciones => Set<Liquidacion>();
    public DbSet<FaltaEmpleado> FaltasEmpleado => Set<FaltaEmpleado>();
    public DbSet<AsignacionEmpleadoTrabajo> Asignaciones => Set<AsignacionEmpleadoTrabajo>();
    public DbSet<Categoria> Categorias => Set<Categoria>();
    public DbSet<CajaMovimiento> CajaMovimientos => Set<CajaMovimiento>();
    public DbSet<Moneda> Monedas => Set<Moneda>();
    public DbSet<Adjunto> Adjuntos => Set<Adjunto>();

    public AppDbContext(DbContextOptions<AppDbContext> options)
        : base(options)
    {
    }

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        modelBuilder.ApplyConfigurationsFromAssembly(typeof(AppDbContext).Assembly);
    }
}
```

# ⚙️ 2. CONFIGURACIONES POR ENTIDAD

Se usa `IEntityTypeConfiguration<T>` → limpio y escalable.

---

## 🔷 2.1 MovimientoConfiguration

```csharp
public class MovimientoConfiguration : IEntityTypeConfiguration<Movimiento>
{
    public void Configure(EntityTypeBuilder<Movimiento> builder)
    {
        builder.ToTable("Movimientos");

        builder.HasKey(x => x.Id);

        builder.Property(x => x.Descripcion)
            .HasMaxLength(500);

        builder.Property(x => x.Cantidad)
            .HasConversion<double>();

        builder.Property(x => x.MontoUnitario)
            .HasConversion<double>();

        builder.Property(x => x.Total)
            .HasConversion<double>();

        builder.Property(x => x.TipoCambio)
            .HasConversion<double>();

        builder.HasOne(x => x.Categoria)
            .WithMany()
            .HasForeignKey(x => x.CategoriaId)
            .OnDelete(DeleteBehavior.Restrict);

        builder.HasOne(x => x.Trabajo)
            .WithMany(t => t.Movimientos)
            .HasForeignKey(x => x.TrabajoId)
            .OnDelete(DeleteBehavior.SetNull);

        builder.HasOne(x => x.Cliente)
            .WithMany()
            .HasForeignKey(x => x.ClienteId)
            .OnDelete(DeleteBehavior.SetNull);

        builder.HasOne(x => x.Empleado)
            .WithMany()
            .HasForeignKey(x => x.EmpleadoId)
            .OnDelete(DeleteBehavior.SetNull);

        builder.HasOne(x => x.Factura)
            .WithMany()
            .HasForeignKey(x => x.FacturaId)
            .OnDelete(DeleteBehavior.SetNull);

        builder.HasIndex(x => x.Fecha);
        builder.HasIndex(x => x.TrabajoId);
    }
}
```

SQLite no soporta decimal real → ya resolvimos con:

```csharp
.HasConversion<double>()
```

# ⚠️ 5. DECISIONES CRÍTICAS YA RESUELTAS

---

## ✔ Eliminaciones seguras

* `Restrict` en relaciones críticas
* `SetNull` donde no rompe lógica
* `Cascade` solo en dependencias claras

---

## ✔ Índices estratégicos

* Fecha (movimientos)
* FacturaId (pagos)
* TrabajoId

---

## ✔ Polimorfismo simple

Adjuntos sin FK directa → más flexible
